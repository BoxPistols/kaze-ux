        background-color: #18181b !important;
      }

      [data-testid="docs-container"] h1,

## Usage

### MUI Components

```tsx
import { useTheme } from '@mui/material/styles'

function MyComponent() {
  const theme = useTheme()
  return (
    <Box
      sx={{
        backgroundColor: theme.palette.background.paper,
        color: theme.palette.primary.main,
      }}>
      Dark Mode Component
    </Box>
  )
}
```

### Tailwind CSS Classes

```tsx
function MyComponent() {
  return (
    <div className='dark:bg-zinc-900 dark:text-primary-main'>
      Dark Mode Component
    </div>
  )
}
```

### Shared Usage

Both MUI and Tailwind use the same color tokens from `colorData.dark`, ensuring design consistency across all components in dark mode.
